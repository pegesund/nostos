Asynchronous I/O & HTTP

Nostos provides fully asynchronous I/O operations for network and file access. You write code as if it were synchronous and blocking, but the runtime transparently handles scheduling other processes while I/O is in progress, ensuring high concurrency and responsiveness.

HTTP Client

The built-in Http module allows making various web requests. These are non-blocking, meaning your program won't freeze while waiting for a response.


fetch_ip() = {
    # Perform a GET request to get public IP
    # I/O operations throw exceptions on error - use try/catch to handle
    try {
        resp = Http.get("https://httpbin.org/ip")
        println("Your IP: " ++ resp.body)
    } catch { e ->
        println("Error fetching IP: " ++ e)
    }
}

main() = {
    fetch_ip()
    println("Client operations are non-blocking!")
}

Parallel HTTP Requests

Leverage Nostos's lightweight processes to perform many HTTP requests concurrently. The runtime ensures efficient use of resources without complex async/await keywords.


timed_request_worker(parent, id) = {
    println("Worker " ++ show(id) ++ " starting request...")
    # This HTTP call yields the process, allowing others to run
    # Exceptions are caught and reported
    try {
        data = Http.get("https://httpbin.org/delay/1")
        println("Worker " ++ show(id) ++ " done: status " ++ show(data.status))
        parent <- ("done", id)
    } catch { e ->
        println("Worker " ++ show(id) ++ " error: " ++ e)
        parent <- ("done", id)
    }
}

main() = {
    me = self()
    println("Spawning 5 HTTP requests with 1-second server-side delay each.")
    println("Total time will be ~1-2 seconds, not 5 seconds, due to parallelism.")

    # Spawn 5 workers; all requests are initiated almost simultaneously
    for i = 1 to 5 {
        spawn { timed_request_worker(me, i) }
    }

    # Collect results from all workers
    for i = 1 to 5 {
        receive ("done", id) -> println("Request from worker " ++ show(id) ++ " completed.")
    }
    println("All parallel requests finished!")
}

HTTP Server

Build scalable web services with Nostos's non-blocking HTTP server. The spawn-per-request pattern spawns a lightweight process for each incoming request, enabling high concurrency.


use stdlib.server.{serve}

handle(req) = {
    match req.path {
        "/" -> Server.respond(req.id, 200, [], "Hello from Nostos!")
        "/echo" -> Server.respond(req.id, 200, [], req.body)
        _ -> Server.respond(req.id, 404, [], "Not Found: " ++ req.path)
    }
}

main() = {
    println("Server running on http://localhost:8080")
    serve(8080, handle)
}

The serve(port, handler) function handles the accept loop internally, spawning a new lightweight process for each request.

Query Parameters

URL query parameters are automatically parsed and available via req.queryParams. This is a list of (key, value) tuples, where values are automatically URL-decoded.


# Helper to find a parameter by name
findParam(params, key) = {
    if length(params) == 0 then ""
    else {
        (k, v) = head(params)
        if k == key then v else findParam(tail(params), key)
    }
}

handle_search(req) = {
    # GET /search?q=hello&limit=10
    query = findParam(req.queryParams, "q")
    limitStr = findParam(req.queryParams, "limit")
    limit = if limitStr == "" then "10" else limitStr

    if query == "" then {
        Server.respond(req.id, 400, [], "Missing 'q' parameter")
    } else {
        response = "Searching for: " ++ query ++ " (limit: " ++ limit ++ ")"
        Server.respond(req.id, 200, [("Content-Type", "text/plain")], response)
    }
}

Cookies

Read cookies from incoming requests via req.cookies, and set cookies in responses using the Set-Cookie header.


handle_login(req) = {
    user = findParam(req.queryParams, "user")
    if user == "" then {
        Server.respond(req.id, 400, [], "Missing 'user' parameter")
    } else {
        # Set a session cookie
        cookieHeader = "session=user_" ++ user ++ "; Path=/; HttpOnly; Max-Age=3600"
        Server.respond(req.id, 200,
            [("Content-Type", "text/plain"), ("Set-Cookie", cookieHeader)],
            "Welcome " ++ user ++ "!")
    }
}

handle_profile(req) = {
    # Read session cookie
    session = findParam(req.cookies, "session")
    if session == "" then {
        Server.respond(req.id, 401, [], "Not logged in")
    } else {
        Server.respond(req.id, 200, [], "Profile page. Session: " ++ session)
    }
}

Form Parameters (POST Data)

For application/x-www-form-urlencoded POST requests, form data is automatically parsed into req.formParams. Values are URL-decoded automatically.


handle_form_submit(req) = {
    if req.method != "POST" then {
        Server.respond(req.id, 405, [], "Use POST method")
    } else {
        name = findParam(req.formParams, "name")
        email = findParam(req.formParams, "email")
        message = findParam(req.formParams, "message")

        if name == "" then {
            Server.respond(req.id, 400, [], "Name is required")
        } else if email == "" then {
            Server.respond(req.id, 400, [], "Email is required")
        } else {
            # Process the form data...
            println("Received message from " ++ name ++ " (" ++ email ++ ")")
            Server.respond(req.id, 200, [],
                "Thank you for your message, " ++ name ++ "!")
        }
    }
}

URL Routing with Path Parameters

Use Server.matchPath to match URL patterns with path parameters (like :id). It returns a list of (paramName, value) tuples if the path matches, or an empty list if it doesn't.


route(req) = {
    path = req.path

    # Static routes
    if path == "/" then handleHome(req)
    else if path == "/api/health" then handleHealth(req)

    # Dynamic routes - try most specific patterns first
    else {
        # Check /users/:userId/posts/:postId (2 params)
        userPostParams = Server.matchPath(path, "/users/:userId/posts/:postId")
        if length(userPostParams) == 2 then {
            # Deconstruct params (returned in order of appearance)
            (_, userId) = head(userPostParams)
            (_, postId) = head(tail(userPostParams))
            handleUserPost(req, userId, postId)
        }

        # Check /users/:userId (1 param)
        else {
            userParams = Server.matchPath(path, "/users/:userId")
            if length(userParams) == 1 then {
                (_, userId) = head(userParams)
                handleUser(req, userId)
            }

            # 404 for anything else
            else {
                Server.respond(req.id, 404, [], "Not Found: " ++ path)
            }
        }
    }
}

WebSockets

Nostos supports WebSocket connections for real-time bidirectional communication. Check if a request is a WebSocket upgrade with req.isWebSocket, then use the WebSocket functions to communicate.


use stdlib.server.{serve}

wsEchoLoop(ws) = {
    ok = try {
        msg = WebSocket.recv(ws)
        if msg == "quit" then {
            WebSocket.send(ws, "Goodbye!")
            WebSocket.close(ws)
            false
        } else {
            WebSocket.send(ws, "Echo: " ++ msg)
            true
        }
    } catch { _ ->
        WebSocket.close(ws)
        false
    }
    if ok then wsEchoLoop(ws) else ()
}

route(req) = {
    if req.isWebSocket then {
        ws = WebSocket.accept(req.id)
        if req.path == "/echo" then wsEchoLoop(ws)
        else {
            WebSocket.send(ws, "Unknown endpoint")
            WebSocket.close(ws)
        }
    } else if req.path == "/" then {
        Server.respond(req.id, 200, [], "WebSocket server. Connect to /echo")
    } else {
        Server.respond(req.id, 404, [], "Not Found")
    }
}

main() = {
    println("WebSocket echo at ws://localhost:8080/echo")
    serve(8080, route)
}

WebSocket API:

  • req.isWebSocket - Check if request is a WebSocket upgrade
  • WebSocket.accept(reqId) - Accept upgrade, returns WebSocket handle
  • WebSocket.send(ws, message) - Send a text message
  • WebSocket.recv(ws) - Receive a message (blocks until available)
  • WebSocket.close(ws) - Close the connection
